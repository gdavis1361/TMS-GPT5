  513.      * @param bool $vIsContractedRate is this pre_order a contracted rate
  514.      *
  515.      * @return bool (true = success false = failure)
  516.      */
  517.     public function update_charges( $nFuelCharge, $nLineHaulCharge, $aAccessorials, $vIsContractedRate ){
  518.         $nPreOrderId = $this->get_pre_order_id();
  519.         if( empty( $nPreOrderId ) ) return false;
  520.         $vSuccess = true;
  521.         $nUserId = get_user_id();
  522.         $nTotalAccessorialCharge = 0;
  523.         $oPreOrderCharge = new PreOrderCharge();
  524.         $oPreOrderCharge->load( $nPreOrderId );
  525.         if( $vIsContractedRate ){
  526.             // linehaul and fuel cannot be changed on contracted_rates
  527.             if( $oPreOrderCharge->get_fuel_charge() > 0 ) // unless there is currently nothing saved
  528.                 $nFuelCharge = $oPreOrderCharge->get_fuel_charge();
  529.             if( $oPreOrderCharge->get_linehaul_charge() > 0 ) // unless there is currently nothing saved
  530.                 $nLineHaulCharge = $oPreOrderCharge->get_linehaul_charge();
  531.         }elseif( time() - strtotime( $this->get_created_at() ) > 432000 ){
  532.             // linehaul cannot be changed after 5 days
  533.             if( $oPreOrderCharge->get_linehaul_charge() > 0 ) // unless there is currently nothing saved
  534.                 $nLineHaulCharge = $oPreOrderCharge->get_linehaul_charge();
  535.         }

  538.         $aAccessorials = reindex_by_array_element( $aAccessorials, 'id' );

  540.         $oAccessorials = new PreOrderAccessorials();
  541.         $oAccessorials->where( 'pre_order_id', '=', $nPreOrderId );
  542.         // delete accessorials missing from sent

  545.                 $oAccessorials->delete( $a->get_order_accessorial_id() );
  546.         }
  547.         // update the remaining existing accessorials
  548.         foreach( $aAccessorials as $nIndex => $a ){
  549.             $oAccessorials = new PreOrderAccessorials();
  550.             if( $a['id'] > 0 )
  551.                 $oAccessorials->load( $a['id'] );
  552.             if( $a['bill_separate'] == 0 ) // don't bill separte

  554.             else // bill separate if have an id to bill to
  555.                 $nBillTo = empty( $a['bill_to_id'] ) ? $this->get_bill_to_id() : $a['bill_to_id'];
  556.             $nCharge = round( $a['per_unit'] * $a['unit_count'], 4);

  558.             $vSuccess = $oAccessorials->create( $nPreOrderId, $a['type_id'], $a['unit_count'], $a['per_unit'], 
  559.                                     $nCharge, $nIndex, $nBillTo, $nUserId );
  560.             if( !$vSuccess ) return false;
  561.         }
  562.         // update my order_charge
  563.         return $oPreOrderCharge->create( $nPreOrderId, $nFuelCharge, $nLineHaulCharge, 
  564.                                     $nTotalAccessorialCharge, $nUserId );
  565.     }
